  const h=parseFloat(cm)/100, p=parseFloat(kg);
  if(!h||!p||isNaN(h)||isNaN(p)) return null;

          <div class="vitals-grid">
            <div class="vital-tile"><div class="vital-icon">💉</div><div class="vital-lbl">TA</div><div class="vital-val">${escH(c.examenFisico?.ta||'—')}</div></div>
            <div class="vital-tile"><div class="vital-icon">🫀</div><div class="vital-lbl">FC</div><div class="vital-val">${c.examenFisico?.fc?c.examenFisico.fc+' lpm':'—'}</div></div>
            <div class="vital-tile"><div class="vital-icon">🫁</div><div class="vital-lbl">Sat O₂</div><div class="vital-val">${c.examenFisico?.saturacion?c.examenFisico.saturacion+'%':'—'}</div></div>
            <div class="vital-tile"><div class="vital-icon">⚖️</div><div class="vital-lbl">IMC</div><div class="vital-val">${vi||'—'}</div><div class="vital-sub" style="color:${imcColor(vi)}">${imcLabel(vi)}</div></div>

    });

  document.getElementById('detail-content').innerHTML = html;

function saveConsulta() {
  const fecha = document.getElementById('c-fecha').value;
  const motivo = document.getElementById('c-motivo').value.trim();
  if (!fecha || !motivo) return showToast('Fecha y motivo son obligatorios', true);
  const c = {
    id: Date.now(), fecha, motivoConsulta: motivo,
    estudiosRealizados: [...newCEsts],
    enfermedadActual: document.getElementById('c-ea').value.trim(),
    claseFuncional: newCCF||'',
    examenFisico: {
      fc: document.getElementById('ef-fc').value,
      ta: document.getElementById('ef-ta').value,
      saturacion: document.getElementById('ef-sat').value,
      peso: document.getElementById('ef-peso').value,
      altura: document.getElementById('ef-alt').value,
      hallazgos: document.getElementById('c-ef').value.trim(),
    },
    planSolicitado: [...newCPlan],
    planNotas: document.getElementById('c-plan').value.trim(),
  };
  const p = patients.find(x=>x.id===currentPatientId);
  if (p) { p.consultas.push(c); save(); }
  showToast('Consulta guardada');
  goBack('detail');
}

// ── MODAL ──────────────────────────────────────────────────────────────────
function openModal(consultId) {
  const p = patients.find(x=>x.id===currentPatientId);
  if (!p) return;
  const c = p.consultas.find(x=>x.id===consultId);
  if (!c) return;
  document.getElementById('modal-title').textContent = c.motivoConsulta||'Consulta';
  document.getElementById('modal-sub').textContent   = c.fecha;
  const vi = calcIMC(c.examenFisico?.peso, c.examenFisico?.altura);

  let html = '';

  // Estudios
  if ((c.estudiosRealizados||[]).length>0) {
    html+=`<div class="modal-block">
      <div class="modal-block-hdr">📄 Estudios realizados</div>
      ${c.estudiosRealizados.map(e=>`
        <div class="est-item">
          <div class="item-body">
            <div class="item-name">${escH(e.tipo)}${e.fecha?' — '+escH(e.fecha):''}</div>
            ${e.resultado?`<div class="item-sub">${escH(e.resultado)}</div>`:''}
          </div>
        </div>`).join('')}
    </div>`;
  }

  // Motivo
  html+=`<div class="modal-block">
    <div class="modal-block-hdr">💬 Motivo de consulta</div>
    <p style="font-size:14px;line-height:1.6;color:var(--ink)">${escH(c.motivoConsulta)}</p>
  </div>`;

  // EA
  if (c.enfermedadActual) {
    html+=`<div class="modal-block">
      <div class="modal-block-hdr">📝 Enfermedad actual</div>
      <p style="font-size:14px;line-height:1.65;color:var(--ink);white-space:pre-wrap">${escH(c.enfermedadActual)}</p>
      ${c.claseFuncional?`<div class="tag tag-blue" style="display:inline-block;margin-top:8px">${escH(c.claseFuncional)}</div>`:''}
    </div>`;
  }

  // Examen físico
  html+=`<div class="modal-block">
    <div class="modal-block-hdr">🩺 Examen físico cardiovascular</div>
    <div class="vitals-grid" style="margin-bottom:12px">
      <div class="vital-tile"><div class="vital-icon">💉</div><div class="vital-lbl">TA</div><div class="vital-val">${escH(c.examenFisico?.ta||'—')}</div></div>
      <div class="vital-tile"><div class="vital-icon">🫀</div><div class="vital-lbl">FC</div><div class="vital-val">${c.examenFisico?.fc?c.examenFisico.fc+' lpm':'—'}</div></div>
      <div class="vital-tile"><div class="vital-icon">🫁</div><div class="vital-lbl">Sat O₂</div><div class="vital-val">${c.examenFisico?.saturacion?c.examenFisico.saturacion+'%':'—'}</div></div>
      <div class="vital-tile"><div class="vital-icon">📏</div><div class="vital-lbl">Peso/Alt</div><div class="vital-val" style="font-size:12px">${c.examenFisico?.peso?c.examenFisico.peso+'kg':'—'} / ${c.examenFisico?.altura?c.examenFisico.altura+'cm':'—'}</div></div>
      <div class="vital-tile"><div class="vital-icon">⚖️</div><div class="vital-lbl">IMC</div><div class="vital-val">${vi||'—'}</div><div class="vital-sub" style="color:${imcColor(vi)}">${imcLabel(vi)}</div></div>
    </div>
    ${c.examenFisico?.hallazgos?`<div style="background:var(--cream);border-radius:10px;padding:12px 14px;font-size:13px;line-height:1.65;color:var(--ink);white-space:pre-wrap">${escH(c.examenFisico.hallazgos)}</div>`:''}
  </div>`;

  // Plan
  if ((c.planSolicitado||[]).length>0 || c.planNotas) {
    html+=`<div class="modal-block">
      <div class="modal-block-hdr">📋 Plan</div>
      ${(c.planSolicitado||[]).length>0?`<div class="tag-row" style="margin-bottom:12px">${c.planSolicitado.map(x=>`<span class="tag tag-blue">${escH(x)}</span>`).join('')}</div>`:''}
      ${c.planNotas?`<div style="background:var(--greenL);border-radius:10px;padding:12px 14px;font-size:13px;line-height:1.65;color:var(--ink);white-space:pre-wrap">${escH(c.planNotas)}</div>`:''}
    </div>`;
  }

  document.getElementById('modal-body').innerHTML = html;
  document.getElementById('modal-overlay').classList.remove('hidden');
  document.body.style.overflow='hidden';
}

function closeModal(e) {
  if (e.target === document.getElementById('modal-overlay')) closeModalDirect();
}
function closeModalDirect() {
  document.getElementById('modal-overlay').classList.add('hidden');
  document.body.style.overflow='';
}
